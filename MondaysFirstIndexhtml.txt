    function applyTranslations() {

    async function init() {
      console.log('🚀 Initialisation SangoCast...');

      const configured = localStorage.getItem('sangocast_configured');
      
      if (!configured || configured !== 'true') {
        console.log('⚠️ Première utilisation - tentative de chargement du wizard...');

        try {
          const probe = await fetch('/data/channels.json', { method: 'HEAD' });
          if (!probe.ok) throw new Error(`channels.json → HTTP ${probe.status}`);
          console.log('✅ channels.json accessible — wizard activé');
          return;
        } catch (err) {
          console.warn('⚠️ channels.json introuvable ou serveur hors ligne:', err.message);
          console.log('🔧 Auto-configuration avec les valeurs par défaut...');

          localStorage.setItem('sangocast_configured',    'true');
          localStorage.setItem('sangocast_channel_id',    'SCR-EN-GL-0001');
          localStorage.setItem('sangocast_bible_versions', JSON.stringify(['KJV']));
          localStorage.setItem('sangocast_age_mode',      'family_mode');
          localStorage.setItem('sangocast_language',      'fr-FR');
          localStorage.setItem('sangocast_tradition',     'protestant');
          localStorage.setItem('sangocast_prayer_times',  JSON.stringify([6, 12, 18, 21]));
          localStorage.setItem('sangocast_install_date',  new Date().toISOString());

          try {
            if (typeof SangocastSetupWizard !== 'undefined') {
              if (typeof SangocastSetupWizard.close   === 'function') SangocastSetupWizard.close();
              if (typeof SangocastSetupWizard.hide    === 'function') SangocastSetupWizard.hide();
              if (typeof SangocastSetupWizard.destroy === 'function') SangocastSetupWizard.destroy();
              if (typeof SangocastSetupWizard.skip    === 'function') SangocastSetupWizard.skip();
            }
          } catch(wizErr) {
            console.warn('Wizard API non disponible:', wizErr.message);
          }

          setTimeout(() => {
            const wizardSelectors = [
              '#setup-wizard', '#wizard', '.setup-wizard', '.wizard-overlay',
              '[id*="wizard"]', '[class*="wizard"]'
            ];
            wizardSelectors.forEach(sel => {
              document.querySelectorAll(sel).forEach(el => {
                el.style.display = 'none';
                console.log('🧹 Wizard masqué:', el.id || el.className);
              });
            });
            document.body.style.overflow = '';
          }, 50);

          sessionStorage.setItem('sangocast_missing_channels', 'true');

          console.log('✅ Valeurs par défaut enregistrées — démarrage normal');
        }
      }

      loadUserConfig();
      applyTranslations();
      await initializeModules();
      startTimers();
      await updateContent();
      displayChannelInfo();

      console.log('✅ SangoCast initialisé');
    }

        await SangocastContentPrioritizer.init(installDate);

        await SangocastHornerEngine.loadPlan();

        SangocastNetworkManager.addChangeListener(updateNetworkIndicator);
        await SangocastNetworkManager.performHeartbeat();

      } catch (error) {
        console.error('❌ Erreur initialisation modules:', error);
      }
    }

    function startTimers() {
      setInterval(updateClock, 1000);
      updateClock();

      setInterval(updateContent, 10 * 60 * 1000);

      setInterval(() => {
        updateNetworkIndicator(SangocastNetworkManager.getStatus());
      }, 5000);

      setInterval(updateLiturgicalBackground, 60 * 60 * 1000);
      updateLiturgicalBackground();

      console.log('⏰ Timers démarrés');
    }

    function updateClock() {
      const now = new Date();
      
      const dateOptions = { 
        weekday: 'long', 
        year: 'numeric', 
        month: 'long', 
        day: 'numeric' 
      };
      const dateStr = now.toLocaleDateString(userConfig.language, dateOptions);
      document.getElementById('current-date').textContent = dateStr;

      const timeStr = now.toLocaleTimeString(userConfig.language, {
        hour: '2-digit',
        minute: '2-digit',
        second: '2-digit'
      });
      document.getElementById('current-time').textContent = timeStr;
    }

    function updateNetworkIndicator(status) {
      const dot = document.getElementById('network-dot');
      const text = document.getElementById('network-status');

      if (status.isOnline) {
        dot.classList.remove('offline');
        text.textContent = t('network-online');
      } else {
        dot.classList.add('offline');
        text.textContent = t('network-offline');
      }
    }

    async function updateContent() {
      try {
        console.log('🔄 Mise à jour contenu...');

        if (typeof SangocastContentPrioritizer === 'undefined') {
          throw new Error('ContentPrioritizer not yet available');
        }

        const content = await SangocastContentPrioritizer.getContentForNow(
          userConfig.channelId,
          userConfig.ageMode,
          userConfig.tradition,
          userConfig.bibleVersion
        );

        currentContent = content;

        const isFallbackVerse =
          content.source === 'fallback' ||
          content.priority === 7 ||
          (content.type === 'text' &&
           content.content &&
           content.content.reference &&
           content.content.reference.includes('23:1') &&
           content.content.text &&
           content.content.text.includes('shepherd'));

        if (isFallbackVerse) {
          console.log('🔄 Priorité 7 détectée — rotation TESTING_PLAN activée');
          startFallbackRotation();
          return;
        }

        stopFallbackRotation();
        displayContent(content);

        console.log('✅ Contenu mis à jour:', content);

      } catch (error) {
        console.error('❌ Erreur mise à jour contenu:', error);
        console.log('📖 SangoCast › Basculement vers le plan de lecture par défaut...');
        startFallbackRotation();
      }
    }

    const TESTING_PLAN = [

    function startFallbackRotation() {
      if (fallbackTimerId) return;
      if (TESTING_PLAN.length === 0) {
        console.warn('⚠️ TESTING_PLAN is empty - no verses to rotate');
        displayError();
        return;
      }
      console.log('⏱️ Rotation TESTING_PLAN démarrée (10 min)');
      displayFallbackVerse();
      fallbackTimerId = setInterval(() => {
        displayFallbackVerse();
      }, 10 * 60 * 1000);
    }

    function stopFallbackRotation() {
      if (fallbackTimerId) {
        clearInterval(fallbackTimerId);
        fallbackTimerId = null;
        console.log('⏹️ Rotation TESTING_PLAN arrêtée — contenu réel disponible');
      }
    }

    /**
     * Build a reference string from a plan entry
     * e.g. "John 3:16" or "Psalm 23:1-3"
     */
    function buildReference(entry) {
      const vv = entry.verses;
      if (vv.length === 1) return `${entry.book} ${entry.chapter}:${vv[0]}`;
      const isRange = vv.every((v, i) => i === 0 || v === vv[i - 1] + 1);
      return isRange
        ? `${entry.book} ${entry.chapter}:${vv[0]}–${vv[vv.length - 1]}`
        : `${entry.book} ${entry.chapter}:${vv.join(',')}`;
    }

    async function displayFallbackVerse() {
      const entry = TESTING_PLAN[fallbackIndex % TESTING_PLAN.length];
      fallbackIndex++;

      const reference = buildReference(entry);
      let verseText = entry.commentary;

      if (typeof BibleLookupModule !== 'undefined') {
        try {
          const result = await BibleLookupModule.lookup({
            book:    entry.book,
            chapter: entry.chapter,
            verses:  entry.verses,
            version: userConfig.bibleVersion || 'KJV'
          });
          if (result && result.text) {
            verseText = result.text;
          }
        } catch (e) {
          console.warn('📖 BibleLookup indisponible, affichage de la référence locale:', reference);
        }
      }

      displayContent({
        type: 'text',
        content: {
          text:        verseText,
          reference:   reference,
          version:     userConfig.bibleVersion || 'KJV',
          commentary:  entry.commentary
        }
      });

      console.log(`📖 Fallback plan › ${reference} [${fallbackIndex}/${TESTING_PLAN.length}]`);
    }

    function updateLiturgicalBackground() {
      const now = new Date();
      const month = now.getMonth() + 1;
      const day = now.getDate();

      let season = 'ordinary';

      if (month === 12 && day >= 1 && day <= 24) {
        season = 'advent';
      }
      else if ((month === 12 && day >= 25) || (month === 1 && day <= 6)) {
        season = 'christmas';
      }
      else if (month === 2 || month === 3) {
        season = 'lent';
      }
      else if (month === 4 || (month === 5 && day <= 15)) {
        season = 'easter';
      }
      else if (month === 5 || month === 6) {
        season = 'pentecost';
      }

      document.body.className = season;

      console.log('🎨 Saison liturgique:', season);
    }

    async function openSettings() {
      try {
        const response = await fetch('/data/channels.json');
        if (response.ok) {
          const data = await response.json();
          availableChannels = data.channels || [];
          
          const channelSelect = document.getElementById('setting-channel');
          channelSelect.innerHTML = availableChannels.map(ch => `
            <option value="${ch.id}" ${ch.id === userConfig.channelId ? 'selected' : ''}>
              ${ch.metadata.name}
            </option>
          `).join('');
        }
      } catch (error) {
        console.warn('Could not load channels:', error);
      }

      document.getElementById('setting-bible').value = userConfig.bibleVersion;
      document.getElementById('setting-age').value = userConfig.ageMode;
      document.getElementById('setting-language').value = userConfig.language;

      document.getElementById('settings-modal').classList.add('active');
    }

    function closeSettings() {
      document.getElementById('settings-modal').classList.remove('active');
    }

    function saveSettings() {
      const newChannelId = document.getElementById('setting-channel').value;
      const newBibleVersion = document.getElementById('setting-bible').value;
      const newAgeMode = document.getElementById('setting-age').value;
      const newLanguage = document.getElementById('setting-language').value;

      localStorage.setItem('sangocast_channel_id', newChannelId);
      localStorage.setItem('sangocast_bible_versions', JSON.stringify([newBibleVersion]));
      localStorage.setItem('sangocast_age_mode', newAgeMode);
      localStorage.setItem('sangocast_language', newLanguage);

      userConfig.channelId = newChannelId;
      userConfig.bibleVersion = newBibleVersion;
      userConfig.ageMode = newAgeMode;
      userConfig.language = newLanguage;

      SangocastAgeFilter.setAgeMode(newAgeMode);
      SangocastPrayerAlarms.setLanguage(newLanguage);

      applyTranslations();

      updateContent();
      displayChannelInfo();

      closeSettings();

      console.log('✅ Paramètres sauvegardés');
    }

        localStorage.removeItem('sangocast_audio_hours');

    }
